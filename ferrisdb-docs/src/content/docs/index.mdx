    ```rust
    // From our actual implementation
    impl MemTable {
        pub fn set(&self, key: Vec<u8>, value: Vec<u8>) -> Result<()> {
            // Simple approach: just insert and track memory
            self.skiplist.insert(key, value);
            self.memory_usage.fetch_add(size, Ordering::Relaxed);
            Ok(())
        }
    }
    ```
